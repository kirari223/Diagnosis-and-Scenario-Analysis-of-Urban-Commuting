def add_scalebar(ax, length=None, x_pos=None, y_pos=None):
    """
    添加比例尺
    
    Args:
        ax: matplotlib axes
        length: 比例尺长度（米）
        x_pos: x位置（0-1）
        y_pos: y位置（0-1）
    """
    length = length or MAP_ELEMENTS['scalebar_length']
    x_pos = x_pos or MAP_ELEMENTS['scalebar_x']
    y_pos = y_pos or MAP_ELEMENTS['scalebar_y']
    
    xlim = ax.get_xlim()
    ylim = ax.get_ylim()
    x_start = xlim[0] + (xlim[1] - xlim[0]) * x_pos
    y_pos_val = ylim[0] + (ylim[1] - ylim[0]) * y_pos
    x_end = x_start + length
    
    segment_len = length / 4
    for i in range(4):
        color = 'white' if i % 2 == 0 else 'black'
        ax.fill([x_start + i*segment_len, x_start + (i+1)*segment_len,
                x_start + (i+1)*segment_len, x_start + i*segment_len],
               [y_pos_val, y_pos_val, y_pos_val - length*0.015, y_pos_val - length*0.015],
               color=color, edgecolor='black', linewidth=0.5)

def add_scalebar(ax, length, x_pos, y_pos, bar_height_ratio=0.08):
    # 读取当前坐标轴 x 方向范围
    xlim = ax.get_xlim()
    # 读取当前坐标轴 y 方向范围
    ylim = ax.get_ylim()
    # 按相对位置计算比例尺起点 x 坐标
    x_start = xlim[0] + (xlim[1] - xlim[0]) * x_pos
    # 按相对位置计算比例尺上边 y 坐标
    y_pos_val = ylim[0] + (ylim[1] - ylim[0]) * y_pos
    # 根据长度计算比例尺终点 x 坐标
    x_end = x_start + length

    # [修改] 比例尺矩形厚度（可调）
    bar_height = length * bar_height_ratio
    # 将比例尺分为 4 段
    segment_len = length / 4

    # 逐段绘制黑白相间矩形
    for i in range(4):
        # 偶数段白色，奇数段黑色
        color = 'white' if i % 2 == 0 else 'black'
        # 绘制当前分段的矩形面
        ax.fill([x_start + i*segment_len, x_start + (i+1)*segment_len,
                x_start + (i+1)*segment_len, x_start + i*segment_len],
               [y_pos_val, y_pos_val, y_pos_val - bar_height, y_pos_val - bar_height],
               color=color, edgecolor='black', linewidth=1)

    # 绘制比例尺上边框
    ax.plot([x_start, x_end], [y_pos_val, y_pos_val], 'k-', linewidth=1)
    # 绘制比例尺下边框
    ax.plot([x_start, x_end], [y_pos_val - bar_height, y_pos_val - bar_height], 'k-', linewidth=1)
    # 绘制比例尺左边框
    ax.plot([x_start, x_start], [y_pos_val, y_pos_val - bar_height], 'k-', linewidth=1)
    # 绘制比例尺右边框
    ax.plot([x_end, x_end], [y_pos_val, y_pos_val - bar_height], 'k-', linewidth=1)

    # 长度>=1000 时用 km 显示
    if length >= 1000:
        label = f'{length/1000:.0f} km'
    # 否则用 m 显示
    else:
        label = f'{length:.0f} m'

    # [修改] 将比例尺数字下移（0.08 -> 0.16）
    label_y = y_pos_val - length * 0.16
    # 在比例尺下方绘制文字标签
    ax.text((x_start + x_end) / 2, label_y, label,
            ha='center', va='top',
            # [修改] 比例尺注释字号放大两倍
            fontsize=LAYOUT_CONFIG['annotation_fontsize'] * 2,
            fontweight='bold', color='black',
            bbox=dict(boxstyle='round,pad=0.2', facecolor='white', edgecolor='none', alpha=0.9))
